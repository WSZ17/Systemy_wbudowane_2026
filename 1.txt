#include "xc.h"

int main(void) {
    
    unsigned char portValue = 0xFFFF; //deklaracja świecą wszystkie diody 
    
    AD1PCFG = 0xFFFF; //port A ustawiony na cyfrowy, port A ma wszystkie diody
    
    TRISA = 0x0000; //TRIS nadanie wejścia i wyjścia, ustawiamy port A na wyjście
    
    LATA = portValue; //LATx, gdzie X to port 
    
    return 0;
}

#include "xc.h"
#include "libpic30.h" //biblioteka do opóźnień

int main(void) {
    
    unsigned char portValue; //deklaracja świecą wszystkie diody 
    
    AD1PCFG = 0xFFFF; //port A ustawiony na cyfrowy, port A ma wszystkie diody
    
    TRISA = 0x0000; //TRIS nadanie wejścia i wyjścia, ustawiamy port A na wyjście
    
    while(1){
        portValue = 0x55; //diody 0 2 4 6
        LATA = portValue;
        
        __delay32(1000000);
        
        portValue = 0xAA; //diody 1 3 5 7
        
        LATA = portValue;
        
        __delay32(1000000);
        
        
    }
    
    return -1;
}
