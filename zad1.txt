int main(void) {

    
    unsigned char portValue;

    AD1PCFG = 0xFFFF;   // port A jako cyfrowy
    TRISA = 0x0000;     // port A jako wyjście

    while(1){
        static unsigned char numer = 0x0F; // 00000111 (3 bity)
        static int kierunek = 1; // 1 = w lewo, -1 = w prawo

        portValue = numer;
        LATA = portValue;

        if(kierunek == 1){
            numer <<= 1;

            // jeśli doszliśmy do lewej krawędzi (np. 11100000)
            if(numer == 0xF0){
                kierunek = -1;
            }

        } else {
            numer >>= 1;

            // jeśli doszliśmy do prawej krawędzi (00000111)
            if(numer == 0x0F){
                kierunek = 1;
            }
        }

        __delay32(1000000);
    }
    
    return -1;
}
